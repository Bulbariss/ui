## Import

```js lineNumbers=true
import { ContactForm } from "@bulbariss/ui";
```

## Usage

```jsx lineNumbers=true live=true
function MyComponent() {
  const colors = {
    themeBgColor: "bg-indigo-600",
    errorBgColor: "bg-red-500",
    errorBorderColor: "border-red-500",
    successBgColor: "bg-green-500",
    loadingColor: "bg-blue-500",
    btnHover: "bg-indigo-700",
    inputColor: "bg-light",
    inputFocusColor: "focus:border-indigo-600",
    inputBorderColor: "border-light",
  };

  const buttonText = {
    defaultText: "Отправить",
    errorText: "Ошибка",
    successText: "Отправлено",
    loadingtext: "",
  };

  return (
    <ContactForm
      className="w-full max-w-md mx-auto"
      buttonText={buttonText}
      colors={colors}
    />
  );
}

render(<MyComponent />);
```

| Name        | Type     | Default                                                                   | Description            |
| ----------- | -------- | ------------------------------------------------------------------------- | ---------------------- |
| `style`     | `string` | `outline-none appearance-none shadow-none box-border align-middle w-full` | Default input styling. |
| `htmlFor`   | `string` |                                                                           | HTMLfor value.         |
| `className` | `string` |                                                                           | className string.      |
| `label`     | `string` |                                                                           | Label text.            |
