## Import

```js lineNumbers=true
import { useIntersect } from "@bulbariss/ui";

const [ref, entry] = useIntersect({
  threshold: 0,
});
```

## Usage

```jsx lineNumbers=true
function MyComponent() {
  const [ref, entry] = useIntersect({
    threshold: 0.2,
  });

  return (
    <div className="h-20 w-20" ref={ref}>
      <p className="font-extrabold text-6xl">
        {entry.isIntersect ? "Hidden" : "Visible"}
      </p>
    </div>
  );
}

render(<MyComponent />);
```
